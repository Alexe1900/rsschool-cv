function Event() {
  this.handlers = [];
}

Event.prototype.subscribe = function(f) {
  if(!(f in this.handlers)) {
    this.handlers.push(f);
  }
}

Event.prototype.unsubscribe = function(f) {
  for(let i = 0; i < this.handlers.length; i++) {
    if (this.handlers[i] == f) {
      this.handlers.splice(i, 1);
    }
  }
}

Event.prototype.emit = function(...args) {
  for(let i = 0; i < this.handlers.length; i++) {
    this.handlers[i](...args);
  }
}

let findMedianSortedArrays = function(nums1, nums2) {
  let mergedArray = nums1;

  nums2.forEach((elem) => {
    mergedArray.push(elem);
  });

  mergedArray = mergedArray.sort((a, b) => {
      if(a > b) {
          return 1;
      } else {
          return -1;
      }
  });
    
  if (mergedArray.length % 2 == 1) {
    return mergedArray[(mergedArray.length - 1) / 2];
  } else {
    return (mergedArray[mergedArray.length  / 2] +
    mergedArray[mergedArray.length  / 2 - 1]) / 2;
  }
};
